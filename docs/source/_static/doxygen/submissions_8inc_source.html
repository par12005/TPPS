331 function tpps_submission_update_stats(array &$form_state) {

336  $form_state['stats']['author_count'] = tpps_file_len($form_state['values']['publication']['secondaryAuthors']['file']);
337  }
338 
339  $form_state['stats']['species_count'] = $form_state['values']['organism']['number'];
340  break;
341 
342  case TPPS_PAGE_2:
343  if (empty($form_state['tpps_type']) or $form_state['tpps_type'] == 'tpps') {
344  $start = new DateTime("{$form_state['values']['StartingDate']['year']}-{$form_state['values']['StartingDate']['month']}");
345  $end = new DateTime("{$form_state['values']['EndingDate']['year']}-{$form_state['values']['EndingDate']['month']}");
346  $dur = date_diff($start, $end);
347  $form_state['stats']['duration'] = $dur->format('%y years %m months');
348  }
349  break;
350 
351  case TPPS_PAGE_3:
352  $form_state['stats']['tree_count'] = 0;
353  for ($i = 1; $i <= $form_state['stats']['species_count']; $i++) {
354  $form_state['stats']['tree_count'] += tpps_file_len($form_state['values']['tree-accession']["species-$i"]['file']);
355  $form_state['stats']['tree_count'] += $form_state['values']['tree-accession']["species-$i"]['file-no-header'];
356  if ($form_state['stats']['species_count'] == 1 or empty($form_state['values']['tree-accession']['check'])) {
357  break;
358  }
359  }
360  break;
361 
362  case TPPS_PAGE_4:
363  $form_state['stats']['phenotype_count'] = 0;
364  $form_state['stats']['unique_phenotypes'] = 0;
365  for ($i = 1; $i <= $form_state['stats']['species_count']; $i++) {

373  }
