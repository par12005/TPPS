565 function tpps_create_plusgeno_view() {
566  $local_db = chado_get_db(array('name' => 'local'));
567 
568  $query = db_select('pg_class', 'p')
569  ->fields('p')
570  ->condition('relname', 'plusgeno_view')
571  ->execute();
572  if (!($result = $query->fetchObject())) {
573  // Create plusgeno_view if it does not exist.
574  db_query('create materialized view chado.plusgeno_view as SELECT DISTINCT p.project_id,
575  d.accession,
576  ppub.title,
577  o.species,
578  count(o.stock_id) OVER (PARTITION BY p.project_id) AS tree_count,
579  sum(phen.count) OVER (PARTITION BY p.project_id) AS phen_count,
580  unique_phen.phenotypes_assessed,
581  g.gen_count
582  FROM chado.project_dbxref p
583  JOIN chado.dbxref d ON p.dbxref_id = d.dbxref_id
584  JOIN ( SELECT pp.project_id,
585  pub.title
586  FROM chado.project_pub pp
587  JOIN chado.pub ON pp.pub_id = pub.pub_id) ppub ON p.project_id = ppub.project_id
588  JOIN ( SELECT DISTINCT ps.project_id,
589  (org.genus::text || \' \'::text) || org.species::text AS species,
590  stock.stock_id
591  FROM chado.project_stock ps
592  JOIN chado.stock ON ps.stock_id = stock.stock_id
593  JOIN chado.organism org ON stock.organism_id = org.organism_id) o ON p.project_id = o.project_id
594  LEFT JOIN ( SELECT stock_phenotype.stock_id,
595  count(stock_phenotype.phenotype_id) OVER (PARTITION BY stock_phenotype.stock_id) AS count
596  FROM chado.stock_phenotype) phen ON o.stock_id = phen.stock_id
597  LEFT JOIN ( SELECT DISTINCT pc.project_id,
598  count(pc.name) OVER (PARTITION BY pc.project_id) AS phenotypes_assessed
599  FROM ( SELECT DISTINCT ps.project_id,
600  ph.name
601  FROM chado.phenotype ph
602  JOIN chado.stock_phenotype sp ON ph.phenotype_id = sp.phenotype_id
603  JOIN chado.stock s ON sp.stock_id = s.stock_id
604  JOIN chado.project_stock ps ON s.stock_id = ps.stock_id) pc) unique_phen ON p.project_id = unique_phen.project_id
605  LEFT JOIN ( SELECT DISTINCT genotype_call.project_id,
606  count(genotype_call.genotype_call_id) OVER (PARTITION BY genotype_call.project_id) AS gen_count
607  FROM chado.genotype_call) g ON p.project_id = g.project_id
608  WHERE d.db_id = ' . $local_db->db_id . ';');
609  variable_set('tpps_local_db', $local_db);
610  }
611  else {
612  $matches = array();
613  $definition = db_select('pg_matviews', 'p')
614  ->fields('p', array('definition'))
615  ->condition('matviewname', 'plusgeno_view')
616  ->execute()
617  ->fetchObject()->definition;
618  preg_match('/d\.db_id ?= ?(\d+)\);/', $definition, $matches);
619  variable_set('tpps_local_db', chado_get_db(array('db_id' => (int) $matches[1])));
620  }
621 }
